```python
class SkillMatchingSystem:
    """技能匹配和推荐系统"""
    
    def match_task_to_skills(self, task_description):
        """将任务匹配到用户技能"""
        # 提取任务关键词
        keywords = self._extract_keywords(task_description)
        
        # 匹配技能
        matched_skills = []
        for skill, weight in self.user_skills.items():
            if self._skill_matches_keywords(skill, keywords):
                matched_skills.append({
                    'skill': skill,
                    'weight': weight,
                    'relevance': self._calculate_relevance(skill, keywords)
                })
        
        # 排序和推荐
        recommendations = self._generate_recommendations(matched_skills)
        
        return recommendations
    
    def generate_learning_path(self, target_skill):
        """生成学习路径"""
        # 基于现有技能到目标技能的路径
        current_skills = set(self.user_skills.keys())
        target_skill_set = set([target_skill])
        
        # 查找技能依赖图
        path = self._find_skill_path(current_skills, target_skill_set)
        
        # 生成学习计划
        learning_plan = self._create_learning_plan(path)
        
        return learning_plan
```
